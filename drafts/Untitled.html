library(biaa)

set.seed(123)

sdata = as.matrix(data)
n = dim(sdata)[1]
m = dim(sdata)[2]

k = 3:6
c = 3:6

arch = biaa(as.matrix(data), k, c, nrep = 10, maxiter = 100000)
